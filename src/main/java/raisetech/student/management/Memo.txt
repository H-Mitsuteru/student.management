  public String getstudentInfo() {   // hello 引数に下の言葉がreturn帰って来る

  @PostMapping("/studentInfo")  // Http://localhost:8080/helloに対して実行
  public void setstudentInfo(String name, String age) {   // hello 引数に下の言葉がreturn帰って来る
